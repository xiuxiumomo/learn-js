## 列表类

```
class List {
    constructor() {
        this.listSize = 0; //列表长度
        this.pos = 0; //列表当前位置
        this.dataStore = []; // 初始化一个空数组来保存列表元素
    }
    //添加
    append(element) {
        this.dataStore[this.listSize++] = element;
    }
    //查找
    find(element) {
        for (var i = 0; i < this.dataStore.length; ++i) {
            if (this.dataStore[i] === element) {
                return i
            }
        }
        return -1;
    }
    //移除
    remove(element) {
        var findAt = this.find(element);
        if (findAt > -1) {
            --this.listSize
            this.dataStore.splice(findAt, 1);
            return true
        }
        return false;
    }
    //获取长度
    length() {
        return this.listSize;
    }
    //打印
    toString() {
        return this.dataStore.join('');
    }
    //往指定元素后面插入元素
    insert(element, after) {
        var findAt = this.find(after);
        if (findAt > -1) {
            this.dataStore.splice(findAt + 1, 0, element);
            ++this.listSize;
            return true
        } else {
            return false;
        }
    }
    //情况
    clear() {
        delete this.dataStore;
        this.dataStore = [];
        this.listSize = this.pos = 0;
    }
    //是否包含某个元素
    contains(element) {
        for (var i = 0; i < this.dataStore.length; ++i) {
            if (this.dataStore[i] === element) {
                return true;
            }
        }
        return false;
    }
    //指针放到列表开头
    front() {
        this.pos = 0;
    }
    //指针放到列表结尾
    end() {
        this.pos = this.dataStore.length - 1;
    }
    //指针上移一位
    prev() {
        if (this.pos > 0) {
            --this.pos;
        }
    }
    //指针下移一位
    next() {
        if (this.pos < this.dataStore.length - 1) {
            ++this.pos
        }
    }
    //当前指针位置
    currPos() {
        return this.pos;
    }
    //指针移动端指定位置
    moveTo(pos) {
        if (pos < 0 || pos > this.dataStore.length - 1) {
            return false
        }
        this.pos = pos;
        return true;
    }
    //获取当前元素
    getElement() {
        return this.dataStore[this.pos];
    }

}





var names = new List();
names.append("Clayton");
names.append("Raymond");
names.append("Cynthia");
names.append("Jennifer");
names.append("Bryan");
names.append("Danny");
names.end();
names.next()
console.log(names.getElement())
```

```
class Stack {
    constructor() {
        this.dataSource = [];
        this.top = 0;

    }
    //往栈中添加元素
    push(element) {
        this.dataSource[this.top++] = element;
    }
    //从栈中弹出元素
    pop() {
        return this.dataSource[--this.top];
    }
    //返回栈顶的元素
    peek() {
        return this.dataSource[this.top - 1];
    }
    //栈的长度
    length() {
        return this.top;
    }
    //清空栈
    clear() {
        this.top = 0;
    }
}
//查找回文 aba 或者 1001 对称的数据或字符串
function isWord(str) {
    var s = new Stack();
    for (let i = 0; i < str.length; ++i) {
        s.push(str[i])
    }
    var new_str = '';
    while (s.length() > 0) {
        new_str += s.pop()
    }
    if (new_str == str) {
        return true
    } else {
        return false
    }
}
console.log(isWord('aba'))

//模拟递归
function fact(n) {
    var s = new Stack();
    while (n > 1) {
        s.push(n--);
    }
    var product = 1;
    while (s.length() > 0) {
        product *= s.pop();
    }
    return product;
}
```

```
        this.dataStore.push(element)

    //读取头部元素
    front() {
        return this.dataStore[0]

    //读取尾部元素
    back() {
        return this.dataStore[this.dataStore.length - 1];
    }
    toString() {
        return this.dataStore.join(',');
    }
    empty() {
        return this.dataStore.length === 0
    }
}
```
